30  void updateWorld(WorldObjectList* worldObjectList) {
31  for (std::map<dexrovMsg::intId_t, HapticsShapeSet*>::iterator it=mapShapes.begin(); it!=mapShapes.end(); ++it) {
32  //(*it).second->removeShapeInHapticDevice(*hapticDevice);
33  delete (*it).second;
34  }
35  mapShapes.clear();
36 
37  std::map<dexrovMsg::intId_t, dexrovMsg::ObjectInfo>& objects = worldObjectList->getObjects();
38 
39  // Create all the PositionAttitudeTransform related to the object by their id
40  for (std::map<dexrovMsg::intId_t, dexrovMsg::ObjectInfo>::iterator it=objects.begin(); it!=objects.end(); ++it) {
41 
42  mapShapes[(*it).first] = new dexrovHAPI::HapticsShapeSet((*it).second.objectId);
43 
44  if (!(*it).second.mesh.isNULL) {
45  mapShapes[(*it).first]->push_back( new dexrovHAPI::Mesh((*it).second.mesh) );
46  }
47  if (!(*it).second.primitive.isNULL) {
48  switch((*it).second.primitive.type) {
49  case dexrovMsg::SolidPrimitive::BOX:
50  mapShapes[(*it).first]->push_back( new dexrovHAPI::Box((*it).second.primitive) );
51  break;
52  case dexrovMsg::SolidPrimitive::SPHERE:
53  mapShapes[(*it).first]->push_back( new dexrovHAPI::Sphere((*it).second.primitive) );
54  break;
55  case dexrovMsg::SolidPrimitive::CYLINDER:
56  mapShapes[(*it).first]->push_back( new dexrovHAPI::Cylinder((*it).second.primitive) );
57  break;
58  case dexrovMsg::SolidPrimitive::CONE:
59  std::cerr<<"WARNING: CONE primitive is not implemented in HAPI, no haptic shape set\n";
60  break;
61  default:
62  std::cerr<<"WARNING: UNDEFINED shape is not implemented, no haptic shape set\n";
63  break;
64  }
65  }
66  /*else {
67  std::cerr<<"WARNING: NULL shape cannot be created, no haptic shape set\n";
68  }*/
69  }
70  updateState(worldObjectList);
71  }

73  void updateState(WorldObjectList* worldObjectList) {
74  std::map<dexrovMsg::intId_t, dexrovMsg::Pose>& poses = worldObjectList->getObjectsPoses();
75  for (std::map<dexrovMsg::intId_t, dexrovMsg::Pose>::iterator it=poses.begin(); it!=poses.end(); ++it) {
76  HAPI::Vec3 pos( (*it).second.position.x, (*it).second.position.y, (*it).second.position.z);
77  Quaternion nquat((*it).second.orientation.x, (*it).second.orientation.y, (*it).second.orientation.z, (*it).second.orientation.w);
78  nquat.normalize();
79  //Rotation R(nquat);
80  mapShapes[(*it).first]->setPosition(pos);
81  mapShapes[(*it).first]->setRotation(nquat);
82  }
