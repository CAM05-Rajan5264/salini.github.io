64  void __initializeDevices( bool initializeGenerators ) {
65  xn::NodeInfoList devicesList;
66  XnStatus rc = context.EnumerateProductionTrees(XN_NODE_TYPE_DEVICE, NULL, devicesList);
67  checkXnStatus(rc, "Enumerate devices");
68 
69  size_t devicesListCount=0;
70  for (xn::NodeInfoList::Iterator it = devicesList.Begin(); it != devicesList.End(); ++it) {
71  ++devicesListCount;
72  xn::NodeInfo deviceNode = (*it);
73 
74  std::ostringstream oss;
75  oss<<"Device "<<devicesListCount;
76  deviceNode.SetInstanceName( oss.str().c_str() );
77  rc = context.CreateProductionTree( deviceNode );
78  checkXnStatus(rc, "Create Device");
79 
80  NiteTracker* tracker = createNewTracker(false);
81  tracker->getDeviceByProductionName( oss.str() );
82  if (initializeGenerators) {
83  tracker->initializeDepthGenerator();
84  tracker->initializeUserGenerator();
85  }
86  }
87  if (devicesListCount==0) {
88  //Then no tracker found, create a fake one...
89  std::cerr<<"No tracker found; create a fake one\n";
90  createNewTracker(true);
91  }
92  rc = context.StartGeneratingAll();
93  checkXnStatus(rc, "Start Generating All");
94  }
