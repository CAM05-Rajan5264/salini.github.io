44  if ( jObj.get<jsonxx::Object>("header").get<jsonxx::String>("file type") != "Opensim Skeleton Coordinates Record") {

50  jsonxx::Array& srcCoord = jObj.get<jsonxx::Object>("header").get<jsonxx::Array>("coordinate name");
51  coordMapping.resize( srcCoord.size() );
52  for (size_t idx=0; idx<srcCoord.size(); ++idx) {
53 
54  size_t mIdx = std::distance(mCoord.begin(), std::find(mCoord.begin(), mCoord.end(), srcCoord.get<jsonxx::String>(idx) ));
55  if ( mIdx < mCoord.size() ) {
56  coordMapping[idx] = mIdx;
57  }
58  else {

60  }

64  _record.resize( recArr.size() );
