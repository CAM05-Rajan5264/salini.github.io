115  //for (size_t idx=0; idx<2; ++idx) {

119 
120  if ( contains(possibleConstBody, pointConst->getProperty_body_1().getValue()) ) {
121  constTask.push_back( new Task(
122  bSet.get( pointConst->getProperty_body_1().getValue() ),
123  pointConst->getProperty_location_body_1().getValue(),
124  bSet.get( pointConst->getProperty_body_2().getValue() ),
125  pointConst->getProperty_location_body_2().getValue(),
126  constCoordinate

130  constTask.push_back( new Task(

167  void assemble() {
168  std::cerr<<"===============> ASSEMBLE!!!! <================\n";
169  double currentDeltaXNorm = 0;

171 
172  __initializeQ();
173  for (size_t idx = 0; idx< nMaxIter; ++idx) {

175  __updateTasks();

178  currentDeltaXNorm = deltaX.norm();
179  if( currentDeltaXNorm < tol) { // below tolerance
