56  void updateWorldToScreenMatT() {

138  void organizeTiles(std::vector< std::pair< TextTile*, osg::Vec3> >& TT, bool isRight, bool isTop) {
139  double direction = (isTop) ? 1 : -1;
140  //POSITIONNING TILES
141  for (std::vector< std::pair< TextTile*, osg::Vec3> >::iterator it=TT.begin(); it!=TT.end(); ++it) {
142  if ( isRight ) {
143  (*it).first->displayOnRight(100, direction*10, 10);
144  }
145  else {
146  (*it).first->displayOnLeft(100, direction*10, 10);
147  }
148 
149  double cXmin = (*it).second.x() + (*it).first->getTextBoundingBox().xMin();
150  double cXmax = (*it).second.x() + (*it).first->getTextBoundingBox().xMax();
151  double cYmin = (*it).second.y() + (*it).first->getTextBoundingBox().yMin();
152  double cYmax = (*it).second.y() + (*it).first->getTextBoundingBox().yMax();
153  //new reserach on previous set
154  double totalYmin = 1e+32;
155  double totalYmax = -1e+32;
156  for (std::vector< std::pair< TextTile*, osg::Vec3> >::iterator it2=TT.begin(); it2!=it; ++it2) {
157  double pXmin = (*it2).second.x() + (*it2).first->getTextBoundingBox().xMin() - 5;
158  double pXmax = (*it2).second.x() + (*it2).first->getTextBoundingBox().xMax() + 5;
159  double pYmin = (*it2).second.y() + (*it2).first->getTextBoundingBox().yMin() - 5;
160  double pYmax = (*it2).second.y() + (*it2).first->getTextBoundingBox().yMax() + 5;
161  if ( (pXmin <= cXmin && cXmin <= pXmax) || (pXmin <= cXmax && cXmax <= pXmax) ) {
162  totalYmin = ( pYmin < totalYmin ) ? pYmin : totalYmin;
163  totalYmax = ( pYmax > totalYmax ) ? pYmax : totalYmax;
164  if ( (totalYmin <= cYmin && cYmin <= totalYmax) || (totalYmin <= cYmax && cYmax <= totalYmax) ) {
165  double diff = (isTop) ? (totalYmax - cYmin) : (totalYmin - cYmax);
166  (*it).first->moveAlongY( diff );
167  cYmin += diff;
168  cYmax += diff;
169  }
170  }
171  }
172  }
