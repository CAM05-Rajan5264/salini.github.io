181 void ReplayerControlWidget::updateControl() {
182  double deltaTime = pimpl->chrono.getDeltaTime();
183  if ( pimpl->isPlaying ) {
184 
185  if (pimpl->useRealTime) {
186  pimpl->currentRealTime += deltaTime;
187  if (pimpl->currentRealTime > pimpl->replayer.getTotalTime() ) {
188  pimpl->currentRealTime = 0;
189  pimpl->replayer.gotoAbsoluteFrame( 0 );
190  }
191  else {
192  pimpl->replayer.goToAbsoluteTime( pimpl->currentRealTime );
193  }
194  }
195  else {
196  pimpl->replayer.gotoRelativeFrame( 1 );
197  }
198 
199  pimpl->replayer.updateReplay();
200 
201  setCurrentFrame( pimpl->replayer.getCurrentFrame(), pimpl->replayer.getCurrentTime() );
202 
203  if ( pimpl->replayer.getCurrentFrame() == 0 && !pimpl->ui.loopBtn->isChecked()) {
204  resetToStartState();
205  }
206  }
207 
208 }
