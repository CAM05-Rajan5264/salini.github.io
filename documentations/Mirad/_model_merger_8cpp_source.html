44  void __recursiveAddChildrenBodies(OpenSim::Model& destModel, const OpenSim::Model& srcModel, std::vector< OpenSim::Body* >& remainingBodies, const std::string& parentBodyName) {
45  //get all children
46  std::vector< OpenSim::Body* > children;
47  std::vector< OpenSim::Body* > otherBodies;
48 
49  for (std::vector< OpenSim::Body* >::iterator it=remainingBodies.begin(); it!=remainingBodies.end(); ++it) {
50  if ( (*it)->hasJoint() && (*it)->getJoint().getParentName() == parentBodyName ) {
51  children.push_back( (*it) );
52  }
53  else {
54  otherBodies.push_back( (*it) );
55  }
56  }
57 
58  for (std::vector< OpenSim::Body* >::iterator it=children.begin(); it!=children.end(); ++it) {
59  destModel.addBody( (*it) );
60  __recursiveAddChildrenBodies( destModel, srcModel, otherBodies, (*it)->getName() );
61  }
62  }

198  jset.remove( &joint );
