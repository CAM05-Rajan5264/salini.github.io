54 void __parsePolys( XMLElement* EPolys, Geometry* poly ) {
55 
56  XMLElement* DarrayConnect = NULL;
57  XMLElement* DarrayOffsets = NULL;
58  for (XMLElement* child = EPolys->FirstChildElement(); child != NULL; child = child->NextSiblingElement()) {
59  string cName(child->Attribute("Name"));
60  if (cName == "connectivity")
61  DarrayConnect = child;
62  else if (cName == "offsets")
63  DarrayOffsets = child;
64  }
65  if (!DarrayConnect || ! DarrayOffsets)
66  throw runtime_error("Some arrays connect/offsets are missing");
67 
68  int offset=0, prevOffset=0;
69  int p0,p1,p2,p3;
70  istringstream issConn( DarrayConnect->GetText() );
71  istringstream issOff( DarrayOffsets->GetText() );
72 
73  while(!issOff.eof()) {
74  issOff >> offset;
75  if (offset - prevOffset == 0) {
76  //nothing to do, should be the end of thel ist
77  }
78  else if (offset - prevOffset == 3) {
79  issConn >> p0 >> p1 >> p2;
80  ref_ptr<DrawElementsUInt> tri = new DrawElementsUInt( osg::PrimitiveSet::TRIANGLES, 0 );
81  tri->push_back(p0);
82  tri->push_back(p1);
83  tri->push_back(p2);
84  poly->addPrimitiveSet( tri.get() );
85  }
86  else if (offset - prevOffset == 4) {
87  issConn >> p0 >> p1 >> p2 >> p3;
88  ref_ptr<DrawElementsUInt> quad = new DrawElementsUInt( osg::PrimitiveSet::QUADS, 0 );
89  quad->push_back(p0);
90  quad->push_back(p1);
91  quad->push_back(p2);
92  quad->push_back(p3);
93  poly->addPrimitiveSet( quad.get() );
94  }
95  else {
96  ref_ptr<DrawElementsUInt> pol = new DrawElementsUInt( osg::PrimitiveSet::POLYGON, 0 );
97  for (int idx=0; idx< offset - prevOffset; ++idx) {
98  issConn >> p0;
99  pol->push_back(p0);
100  }
101  poly->addPrimitiveSet( pol.get() );
102  }
103  prevOffset = offset;
104  }
105 
106  ref_ptr<Vec4Array> colors = new Vec4Array();
107  colors->push_back( osg::Vec4(1,1,1,1) ); //white
108  poly->setColorArray( colors.get() );
109  poly->setColorBinding( osg::Geometry::BIND_OVERALL ); //BIND_PER_PRIMITIVE );
110 }

116 ref_ptr<Geode> __readVTPPiece( XMLElement* ElemPiece ) {
117 
118  ref_ptr<Geode> rootGeode = new Geode();
119  ref_ptr<Geometry> poly = new Geometry();

123  string cName(child->Name());
124  if ( cName == "PointData") {
125  __parsePointData( child, poly.get() );

127  else if ( cName == "Points") {
128  __parsePoints( child, poly.get() );
129  }
130  else if ( cName == "Polys") {
131  __parsePolys(child, poly.get() );
132  }
133  else if ( cName == "CellData") {
134 #ifdef _DEBUG
135  std::cerr<<"WARNING: Get 'CellData' element in Piece; not implemented, so skipped...\n";
136 #endif
137  }
138  else if ( cName == "Verts") {
139 #ifdef _DEBUG
140  std::cerr<<"WARNING: Get 'Verts' element in Piece; not implemented, so skipped...\n";
141 #endif
142  }
143  else if ( cName == "Lines") {
144 #ifdef _DEBUG
145  std::cerr<<"WARNING: Get 'Lines' element in Piece; not implemented, so skipped...\n";
146 #endif
147  }
148  else if ( cName == "Strips") {
149 #ifdef _DEBUG

151 #endif
152  }
153  else

155  }
