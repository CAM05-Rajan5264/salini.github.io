81 void OpensimModelCustomSkin::addFromOsimFile(const std::string& inputFileName) {
82  OpenSim::Model model(inputFileName);
83  std::deque< std::string> possibleGeometryPath = getPossibleGeometryPaths(model);
84  const OpenSim::BodySet& bSet = model.getBodySet();
85 
86  addPossibleGeometryPaths( possibleGeometryPath );
87  for (size_t idx=0; idx<(size_t)bSet.getSize(); ++idx) {
88  const OpenSim::Body& obody = bSet.get(idx);
89  //__extractVisualGeometries( bodyNodes[idx], obody );
90  const OpenSim::VisibleObject& displayer = *obody.getDisplayer();
91 
92  osg::ref_ptr<PositionAttitudeTransform> baseNode = new PositionAttitudeTransform();
93 
94  SimTK::Vec3 scale;
95  displayer.getScaleFactors( scale );
96  SimTK::Transform transform = displayer.getTransform();
97  SimTK::Vec3 pos =transform.p();
98  SimTK::Vec4 Qv = transform.R().convertRotationToQuaternion().asVec4();
99  Quat Q(Qv[1], Qv[2], Qv[3], Qv[0] );
100 
101  baseNode->setScale(Vec3(scale[0], scale[1], scale[2]));
102  baseNode->setPosition( Vec3(pos[0], pos[1], pos[2]) );
103  baseNode->setAttitude( Q );
104 
105  const OpenSim::GeometrySet& geoSet = displayer.getGeometrySet();
106  for (int idx=0; idx<geoSet.getSize(); ++idx) {
107  const OpenSim::DisplayGeometry& displayGeo = geoSet.get(idx);
108  osg::ref_ptr<Node> geoNode = miradOsg::readNodeFile( displayGeo.getGeometryFile() );
109 
110  osg::ref_ptr<MatrixTransform> geoBaseNode = new MatrixTransform();
111 
112  geoBaseNode->setMatrix( convertToOSGMatrix( displayGeo.getTransform() ) );
113 
114  geoBaseNode->addChild( geoNode.get() );
115  baseNode->addChild( geoBaseNode.get() );
116  }
117  getOSGBodyNode( obody.getName() )->addChild( baseNode.get() );
118  }
119  removePossibleGeometryPaths( possibleGeometryPath );
120 }

125 void OpensimModelCustomSkin::addFromJsonFile(const std::string& inputFileName) {
126  std::deque< std::string> possibleGeometryPath;
127  possibleGeometryPath.push_back( osgDB::getFilePath(inputFileName) );
128  addPossibleGeometryPaths( possibleGeometryPath );
129 
130  std::ifstream inputFile(inputFileName.c_str());
131  if (! inputFile) {
132  throw std::runtime_error( std::string("Cannot open file: \"") + inputFileName + "\"\n" );
133  }
134  jsonxx::Object jsonGeoObj, jsonGeoDict;
135  jsonGeoObj.parse(inputFile);
136  jsonGeoDict = jsonGeoObj.get<jsonxx::Object>("custom geometries");
137 
138 
139  for (std::map< std::string, jsonxx::Value*>::const_iterator it=jsonGeoDict.kv_map().begin(); it!=jsonGeoDict.kv_map().end(); ++it) {
140 
141  if ( getOSGBodyNodes().find( (*it).first ) != getOSGBodyNodes().end() ) {
142  MatrixTransform* bodyOsgNode = getOSGBodyNode( (*it).first );
143  jsonxx::Array bArr = (*it).second->get<jsonxx::Array>();
144 
145  for (size_t idx=0; idx<bArr.size(); ++idx) {
146  jsonxx::Object gObj = bArr.get<jsonxx::Object>(idx);
147 
148  std::string fileName = gObj.get<jsonxx::String>("file");
149  Vec3 pos = pimpl->str2Vec3( gObj.get<jsonxx::String>("pos") );
150  Quat rot = pimpl->str2Quat( gObj.get<jsonxx::String>("rot") );
151  Vec3 scale = pimpl->str2Vec3( gObj.get<jsonxx::String>("scale") );
152 
153  ref_ptr<Node> geoNode = miradOsg::readNodeFile( fileName );
154  pimpl->addGeometry( bodyOsgNode, geoNode, pos, rot, scale);
155  }
156  }
157  else {
158 #ifdef _DEBUG
159  std::cerr<<"[warning] cannot found osg node with name: \""<< (*it).first <<"\"\n";
160 #endif
161  }
162  }
163  removePossibleGeometryPaths( possibleGeometryPath );
164 }
